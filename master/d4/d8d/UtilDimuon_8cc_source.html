83 void UtilDimuon::CalcVar(const TLorentzVector& p_pos, const TLorentzVector& p_neg, double& mass, double& pT, double& x1, double& x2, double& xF)
84 {
85  const Double_t mp = 0.938;
86  const Double_t ebeam = 120.;

88  const TLorentzVector p_target(0., 0., 0., mp);
89  const TLorentzVector p_cms = p_beam + p_target;
90

92  mass = p_sum.M();
93  pT = p_sum.Perp();

141 void UtilDimuon::Lab2CollinsSoper(const double px1, const double py1, const double pz1,
142  const double px2, const double py2, const double pz2, double& costh, double& phi)
143 {
144  const double m_mu = 0.1056;
145  TLorentzVector mom1;
146  TLorentzVector mom2;
147  mom1.SetXYZM(px1, py1, pz1, m_mu); //Momentum of muon 1 at Laboratory Frame
148  mom2.SetXYZM(px2, py2, pz2, m_mu); //Momentum of muon 2 at Laboratory Frame
149 
150  //Lorentz Boost to Hadron Rest Frame
151  const double m_p = 0.938;
152  const double E_p = 120.0;
153  const double p_p = sqrt(E_p*E_p - m_p*m_p);
154  const double beta = p_p/E_p;
155  mom1.Boost(0,0,-beta); //mom1.Pz is now boosted
156  mom2.Boost(0,0,-beta); //mom2.Pz is now boosted
157 
158  //Calculate costheta
159  TLorentzVector Q = mom1 + mom2;
160  double k1p = mom1.E() + mom1.Pz();
161  double k2p = mom2.E() + mom2.Pz();
162  double k1m = mom1.E() - mom1.Pz();
163  double k2m = mom2.E() - mom2.Pz();
164  costh = 1.0/Q.M()/sqrt(Q*Q + Q.Px()*Q.Px() + Q.Py()*Q.Py())*(k1p*k2m - k1m*k2p);
165 
166  //Calculate tanphi
167  TVector3 Delta(mom1.Px() - mom2.Px(), mom1.Py() - mom2.Py(), mom1.Pz() - mom2.Pz());
168  TVector3 Q3(Q.Px(),Q.Py(),Q.Pz());
169  TVector3 PA(0, 0, p_p);
170  TVector3 R = PA.Cross(Q3);
171  TVector3 RHat = R.Unit();
172  TVector3 QT = Q3;
173  QT.SetZ(0);
174  double tanphi = sqrt(Q*Q + Q.Px()*Q.Px() + Q.Py()*Q.Py()) / Q.M() * (Delta.X()*RHat.X() + Delta.Y()*RHat.Y()) / (Delta.X()*QT.Unit().X() + Delta.Y()*QT.Unit().Y());
175 
176  //Calculate Phi Quadrant
177  double sinth = sin( acos(costh) );
178  double sinphi = (Delta.X()*RHat.X() + Delta.Y()*RHat.Y())/(Q.M()*sinth);
179 
180  //Calculate Phi
181  phi = atan(tanphi);
182  if (tanphi >= 0 && sinphi >= 0) {;} // phi = phi;}
183  else if (tanphi < 0 && sinphi > 0) {phi += TMath::Pi();}
184  else if (tanphi > 0 && sinphi < 0) {phi += TMath::Pi();}
185  else if (tanphi < 0 && sinphi < 0) {phi += 2*TMath::Pi();}
186 }
