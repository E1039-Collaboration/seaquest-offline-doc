52 void FitRTDist::CalcChi2(const double* pars, double& chi2, int& ndf)
53 {
54  RTCurve rtc(m_n_pt);
55  SetRTCurve(pars, &rtc);
56  double dr = rtc.GetRWidth();
57 
58  chi2 = 0;
59  ndf = 0;
60  double r_width = m_h2_RT->GetYaxis()->GetBinWidth(1);
61  for (int it = m_h2_RT->GetNbinsX(); it > 0; it--) {
62  double cont_t = m_h2_RT->Integral(it, it);
63  if (cont_t < 10) continue;
64  double t_cent = m_h2_RT->GetXaxis()->GetBinCenter(it);
65  double r_func = rtc.EvalR(t_cent);
66  for (int ir = m_h2_RT->GetNbinsY(); ir > 0; ir--) {
67  double cont = m_h2_RT->GetBinContent(it, ir);
68  if (cont <= 0) continue;
69  double r_cent = m_h2_RT->GetYaxis()->GetBinCenter(ir);
70  double cont_func = cont_t * r_width * exp( -pow( (r_cent-r_func)/dr, 2 )/2 ) / (sqrt(2*TMath::Pi()) * dr);
71  double chi2p = pow(cont - cont_func, 2) / cont;
72  //cout << "XXX " << it << " " << ir << " : " << dr << " " << cont_t << " " << cont_func << " " << cont << " " << chi2p << endl;
73  chi2 += chi2p;
74  ndf++;
75  }
76  }
