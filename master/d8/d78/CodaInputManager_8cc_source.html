160 int get_hex_bits (unsigned int hexNum, int numBitFromRight, int numBits)
161 {
162  unsigned int hexBits = 0x0;
163  int shift;
164  unsigned int bitwiseand = 0xF;
165  unsigned int eff = 0xF;
166  int i;
167  // Bitwise method. Shift the bits, and use bitwise AND to get the bits we want
168  // Shift the number to get rid of the bits on the right that we want
169  shift = numBitFromRight - numBits + 1;
170  hexBits = hexNum;
171  hexBits = hexBits >> (4 * shift);
172 
173  // Assemble the number that we will use with the above number
174  // in the bitwise AND operation
175  // so, if we want get_hex_bits(hexNum, 3, 2), it will make 0xFF
176  for (i = 1; i < numBits; i++)
177  {
178  bitwiseand += (eff << (4 * i) );
179  }
180 
181  // Do the bitwise AND operation
182  hexBits = hexBits & bitwiseand;
183  return hexBits;
184 }
