375  double y_max = y0_max + fabs(TY_MAX*(z - z0));
376 
377  int elementID_lo = p_geomSvc->getPlaneNElements(chamIDs[i][k]);;

379  if(p_geomSvc->getPlaneType(chamIDs[i][k]) == 1)

383  }
384  else

390
