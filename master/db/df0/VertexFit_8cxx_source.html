411 double VertexFit::findDimuonVertexFast(SRecTrack& track1, SRecTrack& track2)
412 {
413  //Swim both tracks all the way down, and store the numbers
414  TVector3 pos1[NSTEPS_FMAG + NSTEPS_TARGET + 1];
415  TVector3 pos2[NSTEPS_FMAG + NSTEPS_TARGET + 1];
416  TVector3 mom1[NSTEPS_FMAG + NSTEPS_TARGET + 1];
417  TVector3 mom2[NSTEPS_FMAG + NSTEPS_TARGET + 1];
418  track1.swimToVertex(pos1, mom1);
419  track2.swimToVertex(pos2, mom2);
420 
421  int iStep_min = -1;
422  double dist_min = 1E6;
423  int charge1 = track1.getCharge();
424  int charge2 = track2.getCharge();
425  for(int iStep = 0; iStep < NSTEPS_FMAG + NSTEPS_TARGET + 1; ++iStep)
426  {
427  double dist = (pos1[iStep] - pos2[iStep]).Perp();
428  if(dist < dist_min && FMAGSTR*charge1*mom1[iStep].Px() > 0 && FMAGSTR*charge2*mom2[iStep].Px() > 0)
429  {
430  iStep_min = iStep;
431  dist_min = dist;
432  }
433  }
434 
435  if(iStep_min == -1) return Z_DUMP;
436  return pos1[iStep_min].Z();
437 }

544 int VertexFit::findVertex()
545 {
546  int nIter = 0;
547  double _chisq_kalman_prev = 1E6;
548  for(; nIter < _max_iteration; ++nIter)
549  {
550 #ifdef _DEBUG_ON
551  LogInfo("Iteration: " << nIter);
552 #endif
553 
554  _chisq_vertex = 0.;
555  _chisq_kalman = 0.;
556  for(unsigned int j = 0; j < _trkpar_curr.size(); j++)
557  {
558  _node_vertex.resetFlags();
559  //_node_vertex.getMeasurement() = _vtxpar_curr._r.GetSub(0, 1, 0, 0);
560  //_node_vertex.getMeasurementCov() = _vtxpar_curr._cov.GetSub(0, 1, 0, 1);
561  _node_vertex.setZ(_vtxpar_curr._r[2][0]);
562 
563  _kmfit->setCurrTrkpar(_trkpar_curr[j]);
564  if(!_kmfit->fit_node(_node_vertex))
565  {
566 #ifdef _DEBUG_ON
567  LogInfo("Vertex fit for this track failed!");
568 #endif
569  _chisq_kalman = 1E5;
570  break;
571  }
572  else
573  {
574  _chisq_kalman += _node_vertex.getChisq();
575  }
576 
577  updateVertex();
578  }
579 
581 #ifdef _DEBUG_ON
582  LogInfo("At this iteration: ");
583  LogInfo(_vtxpar_curr._r[2][0] << " === " << _node_vertex.getZ() << " : " << _chisq_kalman << " === " << _chisq_vertex << " : " << _vtxpar_curr._r[0][0] << " === " << _vtxpar_curr._r[1][0]);
584 #endif
585  if(_vtxpar_curr._r[2][0] < Z_UPSTREAM || _vtxpar_curr._r[2][0] > Z_DOWNSTREAM)
586  {
587  _chisq_kalman = 1E5;
588  _chisq_vertex = 1E5;
589  break;
590  }
591 
592  if(nIter > 0 && (fabs(_vtxpar_curr._r[2][0] - _node_vertex.getZ()) < _tolerance || _chisq_kalman > _chisq_kalman_prev)) break;
593  _chisq_kalman_prev = _chisq_kalman;
594  }
595 
596  return nIter+1;
597 }

599 void VertexFit::updateVertex()
600 {
601  double p = fabs(1./_node_vertex.getFiltered()._state_kf[0][0]);
602  double px = _node_vertex.getFiltered()._state_kf[1][0];
603  double py = _node_vertex.getFiltered()._state_kf[2][0];
604  double pz = sqrt(p*p - px*px - py*py);
605 
607  TMatrixD H(2, 3);
608  H.Zero();
609 
610  H[0][0] = 1.;
611  H[1][1] = 1.;
612  H[0][2] = -px/pz;
613  H[1][2] = -py/pz;
614 
615  TMatrixD vertex_dummy(3, 1);
616  vertex_dummy.Zero();
617  vertex_dummy[2][0] = _vtxpar_curr._r[2][0];
618 
619  TMatrixD mxy = _node_vertex.getFiltered()._state_kf.GetSub(3, 4, 0, 0);
620  TMatrixD Vxy = _node_vertex.getFiltered()._covar_kf.GetSub(3, 4, 3, 4);
621  TMatrixD S = SMatrix::invertMatrix(Vxy + SMatrix::getABCt(H, _vtxpar_curr._cov, H));
622  TMatrixD K = SMatrix::getABtC(_vtxpar_curr._cov, H, S);
623  TMatrixD zeta = mxy - H*(_vtxpar_curr._r - vertex_dummy);
624  TMatrixD _r_filtered = _vtxpar_curr._r + K*zeta;
625  TMatrixD _cov_filtered = _vtxpar_curr._cov - K*H*(_vtxpar_curr._cov);
626 
627  _chisq_vertex += SMatrix::getAtBC(zeta, S, zeta)[0][0];
628 
629  _vtxpar_curr._r = _r_filtered;
630  _vtxpar_curr._cov = _cov_filtered;
631 }

696 void VertexFit::fillEvaluation()
697 {
698  if(evalTree == nullptr) return;
699 
700  choice_by_kf_eval = -1;
701  choice_by_vx_eval = -1;
702  double chisq_kf_min = 1E6;
703  double chisq_vx_min = 1E6;
704 
705  m_chisq_kf_eval = 0.;
706  m_z_vertex_eval = 0.;
707  for(int i = 0; i < nStart; ++i)
708  {
709  chisq_kf_eval[i] = chisq_km[i];
710  chisq_vx_eval[i] = chisq_vx[i];
711  z_vertex_eval[i] = z_vertex[i];
712  r_vertex_eval[i] = r_vertex[i];
713  z_start_eval[i] = z_start[i];
714 
715  m_chisq_kf_eval += chisq_kf_eval[i];
716  m_z_vertex_eval += z_vertex_eval[i];
717 
718  if(chisq_kf_min > chisq_km[i])
719  {
720  choice_by_kf_eval = i;
721  chisq_kf_min = chisq_km[i];
722  }
723 
724  if(chisq_vx_min > chisq_vx[i])
725  {
726  choice_by_vx_eval = i;
727  chisq_vx_min = chisq_vx[i];
728  }
729  }
730  m_chisq_kf_eval /= nStart;
731  m_z_vertex_eval /= nStart;
732 
733  s_chisq_kf_eval = 0.;
734  s_z_vertex_eval = 0.;
735  if(nStart == 1)
736  {
737  evalTree->Fill();
738  return;
739  }
740 
741  for(int i = 0; i < nStart; ++i)
742  {
743  s_chisq_kf_eval += ((m_chisq_kf_eval - chisq_kf_eval[i])*(m_chisq_kf_eval - chisq_kf_eval[i]));
744  s_z_vertex_eval += ((m_z_vertex_eval - z_vertex_eval[i])*(m_z_vertex_eval - z_vertex_eval[i]));
745  }
746  s_chisq_kf_eval = sqrt(s_chisq_kf_eval/(nStart-1));
747  s_z_vertex_eval = sqrt(s_z_vertex_eval/(nStart-1));
748 
749  evalTree->Fill();
750 }
