100  //int run = m_bg_data->run;

112  EleList* h4b = &m_bg_data->h4b;
113 
114  if (inte_max == 0) return; // Some readout error
115  m_n_evt_used++;
116 
117  m_h1_inte_max->Fill(inte_max);
118 
119  if (m_inte_cut > 0 && inte_max >= m_inte_cut) return;
120  if (h1t->size() == 0 || h1b->size() == 0 ||
121  h2t->size() == 0 || h2b->size() == 0 ||
122  h3t->size() == 0 || h3b->size() == 0 ||
123  h4t->size() == 0 || h4b->size() == 0 ) return;
124  m_n_evt_tb++;
125 
126  RoadMap map_top;
127  RoadMap map_bot;
128  FindAllRoads(h1t, h2t, h3t, h4t, +1, &map_top);
129  FindAllRoads(h1b, h2b, h3b, h4b, -1, &map_bot);
130 
131  bool fired = false;
132  if (map_top.Contain(m_road_map_pos_top) &&
133  map_bot.Contain(m_road_map_neg_bot) ) { // T+B
134  fired = true;
135  for (RoadMap::Iter it = map_top.Begin(); it != map_top.End(); it++) {
136  if (m_road_map_pos_top->Find(it->first)) m_road_map_pos_top->AddBG(it->second);
137  }
138  for (RoadMap::Iter it = map_bot.Begin(); it != map_bot.End(); it++) {
139  if (m_road_map_neg_bot->Find(it->first)) m_road_map_neg_bot->AddBG(it->second);
140  }
141  }
142  if (map_top.Contain(m_road_map_neg_top) &&
143  map_bot.Contain(m_road_map_pos_bot) ) { // B+T
144  fired = true;
145  for (RoadMap::Iter it = map_top.Begin(); it != map_top.End(); it++) {
146  if (m_road_map_neg_top->Find(it->first)) m_road_map_neg_top->AddBG(it->second);
147  }
148  for (RoadMap::Iter it = map_bot.Begin(); it != map_bot.End(); it++) {
149  if (m_road_map_pos_bot->Find(it->first)) m_road_map_pos_bot->AddBG(it->second);
150  }
151  }
152 
153  if (fired) m_n_evt_fired++;
154 }
